## 一、模块概述

漫画生成模块负责：
1. 基于角色和场景描述生成漫画分镜
2. 保持角色一致性
3. 应用镜头控制（推/拉/摇/移/跟）

## 二、生成流程

```
用户输入 → Prompt构建 → 角色LoRA加载 → 图像生成 → 后处理 → 返回结果
```

### 2.1 Prompt构建

```python
def build_prompt(
    scene_description: str,
    characters: List[Character],
    style: str = "日漫",
    camera_type: str = "静止"
) -> str:
    # 1. 角色描述
    char_prompts = []
    for char in characters:
        char_prompt = f"{char.features.hair}, {char.features.eyes}, {char.features.clothing}"
        char_prompts.append(char_prompt)
    
    # 2. 镜头描述
    camera_prompts = {
        "静止": "straight-on shot, front view",
        "推进": "close-up shot, zoom in",
        "拉远": "wide shot, zoom out",
        "摇镜": "pan from left to right",
        "平移": "tracking shot, moving sideways",
        "跟随": "follow cam, tracking subject"
    }
    
    # 3. 风格描述
    style_prompts = {
        "日漫": "anime style, cel shading, vibrant colors",
        "美漫": "american comic style, bold lines, high contrast",
        "国漫": "chinese comic style, ink painting, traditional"
    }
    
    # 组合
    prompt = f"{', '.join(char_prompts)}, {scene_description}, {camera_prompts[camera_type]}, {style_prompts[style]}, high quality, detailed"
    
    return prompt
```

### 2.2 角色LoRA加载

```python
async def load_character_loras(character_ids: List[str]) -> List[str]:
    """加载多个角色LoRA，返回lora路径列表"""
    lora_paths = []
    for char_id in character_ids:
        char = await get_character(char_id)
        if char.lora_path:
            lora_paths.append(char.lora_path)
    return lora_paths

def generate_with_lora(
    prompt: str,
    lora_paths: List[str],
    negative_prompt: str = "low quality, blurry, distorted"
):
    """使用LoRA生成图像"""
    # SD + LoRA推理
    image = pipe.generate(
        prompt=prompt,
        negative_prompt=negative_prompt,
        lora_paths=lora_paths,
        num_inference_steps=30,
        guidance_scale=7.5
    )
    return image
```

### 2.3 异步任务

```python
from celery import Celery

celery_app = Celery('comic_generation')

@celery_app.task
def generate_comic_task(
    character_ids: List[str],
    scene_description: str,
    panel_count: int,
    style: str,
    camera_types: List[str]
):
    """漫画生成异步任务"""
    task_id = generate_comic_task.request.id
    
    # 1. 更新状态
    update_task_status(task_id, "generating")
    
    # 2. 加载LoRA
    lora_paths = load_character_loras(character_ids)
    
    # 3. 分镜生成
    panels = []
    for i in range(panel_count):
        # 构建prompt
        camera = camera_types[i] if i < len(camera_types) else "静止"
        prompt = build_prompt(scene_description, characters, style, camera)
        
        # 生成
        image = generate_with_lora(prompt, lora_paths)
        
        # 保存
        panel = save_panel_image(task_id, i, image)
        panels.append(panel)
        
        # 更新进度
        update_task_progress(task_id, i + 1, panel_count)
    
    # 4. 完成
    result = {"panels": panels, "status": "completed"}
    save_task_result(task_id, result)
    
    return result
```

## 三、数据结构

### 3.1 生成请求

```python
@dataclass
class ComicGenerationRequest:
    character_ids: List[str]      # 角色ID列表
    scene_description: str         # 场景描述
    panel_count: int = 3          # 分镜数量
    style: str = "日漫"           # 画面风格
    camera_types: List[str] = None  # 镜头类型列表
    negative_prompt: str = None   # 负面prompt

@dataclass
class ComicPanel:
    index: int                    # 分镜序号
    image_url: str                # 生成的图片URL
    prompt: str                   # 使用的prompt
    camera_type: str              # 镜头类型
    negative_prompt: str           # 负面prompt
```

### 3.2 数据库表

```sql
CREATE TABLE comic_tasks (
    id VARCHAR(36) PRIMARY KEY,
    status VARCHAR(20) DEFAULT 'pending',  -- pending, generating, completed, failed
    character_ids JSONB,
    scene_description TEXT,
    panel_count INTEGER,
    style VARCHAR(20),
    camera_types JSONB,
    progress INTEGER DEFAULT 0,
    result JSONB,
    error_message TEXT,
    created_at TIMESTAMP DEFAULT NOW(),
    completed_at TIMESTAMP
);

CREATE TABLE comic_panels (
    id VARCHAR(36) PRIMARY KEY,
    task_id VARCHAR(36) REFERENCES comic_tasks(id),
    panel_index INTEGER,
    image_url VARCHAR(255),
    prompt TEXT,
    camera_type VARCHAR(20),
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_tasks_status ON comic_tasks(status);
CREATE INDEX idx_tasks_created ON comic_tasks(created_at);
```

## 四、API接口

### 4.1 创建生成任务

**POST** `/api/comics/generate`

Request:
```json
{
  "character_ids": ["char_001", "char_002"],
  "scene_description": "在咖啡馆里和朋友聊天",
  "panel_count": 3,
  "style": "日漫",
  "camera_types": ["推进", "静止", "拉远"]
}
```

Response:
```json
{
  "task_id": "task_xxx",
  "status": "generating",
  "message": "任务已创建，生成中..."
}
```

### 4.2 查询任务状态

**GET** `/api/comics/tasks/{task_id}`

Response:
```json
{
  "task_id": "task_xxx",
  "status": "generating",
  "progress": 2,
  "total": 3,
  "panels": [
    {"index": 1, "image_url": "https://..."},
    {"index": 2, "image_url": "https://..."}
  ]
}
```

### 4.3 获取生成结果

**GET** `/api/comics/tasks/{task_id}/result`

Response:
```json
{
  "task_id": "task_xxx",
  "status": "completed",
  "panels": [
    {
      "index": 1,
      "image_url": "https://xxx/panel_1.jpg",
      "camera_type": "推进",
      "prompt": "..."
    }
  ],
  "total_panels": 3
}
```
