```python
        "--base-url",
        action="store",
        default="http://localhost:3000",
        help="Base URL for the application",
    )
```

```python
"""
```

```python
from playwright.sync_api import Page, Locator, TimeoutError as PlaywrightTimeoutError
from typing import Optional, List


class ProductPage:
    """商品详情页"""
    
    # ============ 选择器定义 ============
    
    SELECTORS = {
        # 页面加载状态
        "loading": ".ant-skeleton",
        "product_title": "h1",
        "error_state": "[class*='empty']",
        
        # 图片相关
        "main_image": "[class*='imageGallery'] img",
        "image_nav_prev": "[class*='navButton']:first-child",
        "image_nav_next": "[class*='navButton']:last-child",
        "thumbnails": "[class*='thumbnails'] img",
        "image_indicator": "[class*='indicator']",
        
        # 价格相关
        "current_price": "[class*='currentPrice']",
        "original_price": "[class*='originalPrice']",
        "promotion_tags": "[class*='promotionTag']",
        
        # SKU 选择
        "color_options": "[class*='colorOption']",
        "size_options": "[class*='sizeOption']",
        "selected_color": "[class*='colorOption'][class*='selected']",
        "selected_size": "[class*='sizeOption'][class*='selected']",
        "sold_out_badge": "[class*='soldOut']",
        
        # 数量选择
        "quantity_input": "input[type='number']",
        "quantity_minus": "button:has-text('-')",
        "quantity_plus": "button:has-text('+')",
        "stock_info": "[class*='stockInfo']",
        
        # 操作按钮
        "add_to_cart_btn": "button:has-text('加入购物车')",
        "buy_now_btn": "button:has-text('立即购买')",
        "favorite_btn": "[class*='favoriteBtn']",
        
        # Tab 页
        "tab_detail": "button:has-text('商品详情')",
        "tab_spec": "button:has-text('规格参数')",
        "tab_review": "button:has-text('评价')",
        "tab_recommendation": "button:has-text('推荐')",
        
        # 评价相关
        "review_cards": "[class*='reviewCard']",
        "review_stats": "[class*='reviewStats']",
        "filter_tags": "[class*='filterTag']",
        
        # 推荐相关
        "recommendation_grid": "[class*='recommendationGrid']",
        "recommendation_cards": "[class*='recommendationCard']",
        "refresh_btn": "[class*='refreshButton']",
        
        # 提示消息
        "success_message": ".ant-message-success",
        "warning_message": ".ant-message-warning",
        "error_message": ".ant-message-error",
        
        # 底部栏 (移动端)
        "mobile_bottom_bar": "[class*='mobileBottomBar']",
    }
    
    def __init__(self, page: Page):
        self.page = page
    
    # ============ 页面状态检查 ============
    
    def is_loaded(self, timeout: int = 10000) -> bool:
        """检查页面是否加载完成"""
        try:
            self.page.wait_for_selector(self.SELECTORS["product_title"], timeout=timeout)
            return True
        except PlaywrightTimeoutError:
            return False
    
    def is_loading(self) -> bool:
        """检查是否在加载中"""
        return self.page.is_visible(self.SELECTORS["loading"])
    
    def has_error(self) -> bool:
        """检查是否有错误"""
        return self.page.is_visible(self.SELECTORS["error_state"])
    
    def wait_for_load(self, timeout: int = 15000):
        """等待页面加载完成"""
        self.page.wait_for_load_state("networkidle", timeout=timeout)
        if self.is_loading():
            self.page.wait_for_selector(
                self.SELECTORS["product_title"],
                state="visible",
                timeout=timeout
            )
    
    # ============ 商品信息获取 ============
    
    def get_product_title(self) -> str:
        """获取商品标题"""
        return self.page.text_content(self.SELECTORS["product_title"]) or ""
    
    def get_current_price(self) -> str:
        """获取现价"""
        return self.page.text_content(self.SELECTORS["current_price"]) or ""
    
    def get_original_price(self) -> str:
        """获取原价"""
        return self.page.text_content(self.SELECTORS["original_price"]) or ""
    
    def get_promotions(self) -> List[str]:
        """获取促销信息"""
        elements = self.page.query_selector_all(self.SELECTORS["promotion_tags"])
        return [el.text_content() or "" for el in elements]
    
    # ============ 图片操作 ============
    
    def get_main_image_src(self) -> Optional[str]:
        """获取主图 URL"""
        img = self.page.query_selector(self.SELECTORS["main_image"])
        return img.get_attribute("src") if img else None
    
    def click_prev_image(self):
        """点击上一张图片"""
        self.page.click(self.SELECTORS["image_nav_prev"])
    
    def click_next_image(self):
        """点击下一张图片"""
        self.page.click(self.SELECTORS["image_nav_next"])

        """获取缩略图数量"""
        return len(self.page.query_selector_all(self.SELECTORS["thumbnails"]))

        """点击指定缩略图"""
        thumbnails = self.page.query_selector_all(self.SELECTORS["thumbnails"])
        if 0 <= index < len(thumbnails):
            thumbnails[index].click()
    
    # ============ SKU 选择 ============
    
    def get_color_options(self) -> List[str]:
        """获取颜色选项"""
        elements = self.page.query_selector_all(self.SELECTORS["color_options"])
        return [el.get_attribute("title") or el.text_content() or "" for el in elements]
    
    def select_color(self, color: str):
        """选择颜色"""
        color_map = {
            "红色": "[title='红色']",
            "蓝色": "[title='蓝色']",
            "黑色": "[title='黑色']",
            "白色": "[title='白色']",
        }
        selector = color_map.get(color, f"[title='{color}']")
        self.page.click(selector)
```

```python
"""
测试运行脚本
"""
import subprocess
import sys
import os
from pathlib import Path


def run_command(cmd: list, description: str):
    """运行命令并打印结果"""
    print(f"\n{'='*60}")
    print(f"运行: {description}")
    print(f"命令: {' '.join(cmd)}")
    print('='*60)
    
    result = subprocess.run(cmd, capture_output=True, text=True)
    
    print(result.stdout)
    if result.stderr:
        print("STDERR:", result.stderr)
    
    return result.returncode == 0


def main():
    """主函数"""
    # 获取项目根目录
    project_root = Path(__file__).parent.parent
    
    # 切换到项目根目录
    os.chdir(project_root)
    
    all_passed = True
    
    # 1. 运行单元测试
    print("\n" + "="*70)
    print("步骤 1: 运行单元测试")
    print("="*70)
    
    unit_passed = run_command(
        ["pytest", "tests/unit/", "-v", "--tb=short", "-m", "unit"],
        "单元测试"
    )
    all_passed = all_passed and unit_passed
    
    # 2. 运行集成测试
    print("\n" + "="*70)
    print("步骤 2: 运行集成测试")
    print("="*70)
    
    integration_passed = run_command(
        ["pytest", "tests/integration/", "-v", "--tb=short", "-m", "integration"],
        "集成测试"
    )
    all_passed = all_passed and integration_passed
    
    # 3. 运行 E2E 测试（需要启动前端服务）
    print("\n" + "="*70)
    print("步骤 3: 运行 E2E 测试")
    print("="*70)
    
    # 检查是否配置了 base_url
    base_url = os.environ.get("BASE_URL", "http://localhost:3000")
    
    e2e_passed = run_command(
        [
            "pytest", 
            "tests/e2e/", 
            "-v", 
            "--tb=short", 
            "-m", "e2e",
            "--browser=chromium",
            f"--base-url={base_url}"
        ],
        "E2E 测试"
    )
    all_passed = all_passed and e2e_passed
    
    # 总结
    print("\n" + "="*70)
    print("测试总结")
    print("="*70)
    if all_passed:
        print("✅ 所有测试通过!")
        return 0
    else:
        print("❌ 部分测试失败")
        return 1


if __name__ == "__main__":
    sys.exit(main())
```
